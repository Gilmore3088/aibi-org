```ts
export interface Cohort {
  readonly startDate: string;      // ISO 8601 YYYY-MM-DD
  readonly sessionDay: string;     // e.g. "Tuesdays"
  readonly sessionTime: string;    // e.g. "12:00–1:30 PM ET"
  readonly instructor: string;
  readonly enrollDeadline: string; // ISO 8601 YYYY-MM-DD
}

export const cohorts: readonly Cohort[] = [
  { startDate: '2026-05-05', sessionDay: 'Tuesdays', sessionTime: '12:00–1:30 PM ET', instructor: 'The AI Banking Institute', enrollDeadline: '2026-04-28' },
  { startDate: '2026-07-14', ... },
  { startDate: '2026-09-22', ... },
];
```

Getter (new): `src/app/courses/aibi-s/_lib/getUpcomingCohort.ts`
```ts
export function getUpcomingCohort(now: Date = new Date()): Cohort | null {
  const today = now.toISOString().slice(0, 10); // YYYY-MM-DD
  return cohorts.find((c) => c.enrollDeadline >= today) ?? null;
}
```

If the team chooses DB-backed cohorts, create migration:
```sql
CREATE TABLE course_cohorts (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  product text NOT NULL CHECK (product IN ('aibi-s', 'aibi-l')),
  start_date date NOT NULL,
  session_day text NOT NULL,
  session_time text NOT NULL,
  instructor text NOT NULL,
  enroll_deadline date NOT NULL,
  created_at timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX idx_course_cohorts_product_deadline
  ON course_cohorts(product, enroll_deadline);

ALTER TABLE course_cohorts ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Cohorts are publicly readable"
  ON course_cohorts FOR SELECT TO anon, authenticated USING (true);
```
